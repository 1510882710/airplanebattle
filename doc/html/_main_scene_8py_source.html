114 # 碰撞检测
115 def detect_conlision(self):
116 # 敌机和英雄子弹
117 for bullet in self.hero.bullets.bullet_list:
118 if not bullet.visible:
119 continue
120 for enemy in self.enemy.enemies:
121 if not enemy.visible or not bullet.visible:
122 continue
123 # 判断两个边框是否相交 就是 碰撞
124 if pygame.Rect.colliderect(bullet.bbox, enemy.bbox):
125 bullet.set_unused()
126 enemy.set_unused()
127 self.defeat_count += 1
128
129 # 敌机和英雄飞机
130 for enemy in self.enemy.enemies:
131 if not enemy.visible:
132 continue
133 if pygame.Rect.colliderect(self.hero.bbox, enemy.bbox):
134 enemy.set_unused()
135 self.impact_count += 1
136
137 # 敌机子弹和英雄飞机
138 for enemy in self.enemy.enemies:
139 if not enemy.bullet.visible:
140 continue
141 if pygame.Rect.colliderect(self.hero.bbox, enemy.bullet.bbox):
142 enemy.bullet.set_unused()
143 self.damage_count += 1
144
145 # 敌机子弹和英雄子弹
146 for enemy in self.enemy.enemies:
147 if not enemy.bullet.visible:
148 continue
149 for bullet in self.hero.bullets.bullet_list:
150 if not bullet.visible:
151 continue
152 if pygame.Rect.colliderect(bullet.bbox, enemy.bullet.bbox):
153 enemy.bullet.set_unused()
